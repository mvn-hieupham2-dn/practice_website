Question: Let and Const – What's the Difference?

MAX_SIZE = 25 * 1024 * 1024;
var title = 'Hello World';
title = 'Hello ES6'

convert to ES6:
const MAX_SIZE = 25 * 1024 * 1024;
let title = 'Hello World';
title = 'Hello ES6'
difference between let and Const
let có thể cập nhật lại giá trị nhưng không thể khai báo lại trong cùng một phạm vi
let có thể khai báo mà không cần khởi tạo
let không thể truy cập mà không không khởi tạo và sẽ trả về một lỗi
const không thể cập nhật lại giá trị và không thể khai báo lại trong cùng một phạm vi
const không thể khai báo khi không cần khởi tạo
const không thể truy cập mà không không khởi tạo

3. Rest Parameter - ES5
``` function foo(x, y) {
var a = Array.prototype.slice.call(arguments, 2);
return (x + y) * a.length;
};
foo(1, 2, 'hello', true, 7) === 9;
```

- ES6
``` function foo(x,y,...rest) {
return (x + y) * rest.length;
};
foo(1, 2, 'hello', true, 7) === 9;
```

4. Default Parameter Values
- ES5
``` function sum(x, y, z) {
if (y === undefined) {
y = 7;
}
if (z === undefined) {
z = 42;
}
return x + y + z;
};
``` ​

- ES6
``` function sum(x, y=7, z=42) {
if (y === undefined) {
y = 7;
}
if (z === undefined) {
z = 42;
}
return x + y + z;
};

Câu 5: 5. Arrow Functions
- ES5
``` var evens = [1, 2, 3, 4, 5, 6];
var odds = evens.map(function (v) { return v + 1; });
var pairs = evens.map(function (v) { return { even: v, odd: v + 1 }; });
var nums = evens.map(function (v, i) { return v + i; });
var fives = [];
nums.forEach(function (v) {
if (v % 5 === 0) {
fives.push(v);
}
});
``` ​

- ES6
``` let evens = [1, 2, 3, 4, 5, 6];
let odds = evens.map((v) => { return v + 1; });
let pairs = evens.map((v) => { return { even: v, odd: v + 1 }; });
let nums = evens.map((v, i) => { return v + i; });
let fives = [];
nums.forEach((v) => {
if (v % 5 === 0) {
fives.push(v);
}
});

8. Promise
- ES5
``` function showMessAfterTimeout(msg, who, timeout, onDone) {
setTimeout(function () {
onDone(msg + ' Hi ' + who + '!');
}, timeout);
} showMessAfterTimeout('', 'Foo', 100, function(msg) {
showMessAfterTimeout(msg, 'Bar', 200, function(msg) {
console.log('Finish after 300ms:' + msg);
});
});
```

- ES6
``` let showMessAfterTimeout(who, timeout) = () => {
return new Promise((resolve, reject)=>{ return setTimeout(() =>{
onDone(msg + ' Hi ' + who + '!');
}, timeout);
}) } showMessAfterTimeout('Foo', 100)
.then((msg1) => {
return showMessAfterTimeout('Bar', 200).then((msg) => {
return msg1 + ' ' + msg;
});
})
.then((msg) => console.log(`Finish after 300ms: ${msg}`));
